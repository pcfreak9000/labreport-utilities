c func N
c data a
c data b
c data c
sete N "1.2566*10^(-6)*(a-c)/b" a b c
sete c 550 5
sete b 0.578 0.005
sete a 813 8
p -p N a=a b=b c=c
exit

%Show the main help message
-h

%Create various tablets
create func kappa
create data pa
c data pb
c data pc
c data result

%Setting the content of the tablets
sete kappa "k = (a-b)/(a-c)" a b c
setf pa example_data.csv --offset=2 --valueIndex=2 --errorIndex=3 --count=7
setf pb example_data.csv -o=2 -v=4 -e=5 -c=7
setf pc example_data.csv -o=2 -v=6 -e=7 -c=7 

%Calculate the result
propagate kappa -r=result --print a=pa b=pb c=pc

%TeX form stuff
tex kappa -p=linear -v a -v b -v c
tex kappa

%Printing the contents of the tablets
print kappa
print pa
print result

%Exit the program
exit
